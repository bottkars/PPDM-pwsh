                delete (Dell.Metrics.sc.btnname);

                Dell.Metrics.sc['btnname'] = value.btnname;
                Dell.Metrics.sc.pagename = Dell.Metrics.sc.pagename + value.pagename;
            }

                Dell.Metrics.sc['btnname'] = fromSource;
